using Toybox.System;

public function test() {

  // This for loop should only print the values 5, 6, and 7.
  for (var i = 0; i < 10; i += 1) {
    if (i < 5) {
      continue;
    }
    System.println(i);
    if (7 == i) {
      break;
    }
  }
}
